109 /* Compare memory in constant time to mitigate timing attacks.
110  * Returns true if ptr1 and ptr2 are equal for length bytes. */
111 static UA_INLINE UA_Boolean
112 UA_constantTimeEqual(const void *ptr1, const void *ptr2, size_t length) {
113  volatile const UA_Byte *a = (volatile const UA_Byte *)ptr1;
114  volatile const UA_Byte *b = (volatile const UA_Byte *)ptr2;
115  volatile UA_Byte c = 0;
116  for(size_t i = 0; i < length; ++i) {
117  UA_Byte x = a[i], y = b[i];
118  c |= x ^ y;
119  }
120  return !c;
121 }
