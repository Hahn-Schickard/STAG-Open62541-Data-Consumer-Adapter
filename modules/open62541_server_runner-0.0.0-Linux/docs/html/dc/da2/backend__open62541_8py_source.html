45 def sortNodes(nodeset):

47  hasTypeDef = NodeId("ns=0;i=40")

50  if ref.referenceType == hasTypeDef:

52 
53  # Only hierarchical types...

55 
56  # determine in-degree of unfulfilled references
57  L = [node for node in nodeset.nodes.values() if node.hidden] # ordered list of nodes
58  R = {node.id: node for node in nodeset.nodes.values() if not node.hidden} # remaining nodes
59  in_degree = {id: 0 for id in R.keys()}
60  for u in R.values(): # for each node
61  for ref in u.references:
62  if not ref.referenceType in relevant_refs:
63  continue
64  if nodeset.nodes[ref.target].hidden:
65  continue
66  if ref.isForward:
67  continue
68  in_degree[u.id] += 1
69 
70  # Print ReferenceType and DataType nodes first. They may be required even
71  # though there is no reference to them. For example if the referencetype is
72  # used in a reference, it must exist. A Variable node may point to a
73  # DataTypeNode in the datatype attribute and not via an explicit reference.
74 
75  Q = [node for node in R.values() if in_degree[node.id] == 0 and
76  (isinstance(node, ReferenceTypeNode) or isinstance(node, DataTypeNode))]
77  while Q:
78  u = Q.pop() # choose node of zero in-degree and 'remove' it from graph
79  L.append(u)
80  del R[u.id]
81 
82  for ref in u.references:
83  if not ref.referenceType in relevant_refs:
84  continue

86  continue
87  if not ref.isForward:
88  continue

90  if in_degree[ref.target] == 0:
91  Q.append(R[ref.target])
92 
93  # Order the remaining nodes
94  Q = [node for node in R.values() if in_degree[node.id] == 0]
95  while Q:
96  u = Q.pop() # choose node of zero in-degree and 'remove' it from graph
97  L.append(u)
98  del R[u.id]
99

101  if not ref.referenceType in relevant_refs:
102  continue

104  continue

106  continue
107  in_degree[ref.target] -= 1
108  if in_degree[ref.target] == 0:
109  Q.append(R[ref.target])
110

112  for u in nodeset.nodes.values():
113  for ref in u.references:
114  if ref.referenceType == hasTypeDef:
115  ref.isForward = not ref.isForward
116

126  raise Exception("Node graph is circular on the specified references. Still open nodes:\r\n" + stillOpen)
