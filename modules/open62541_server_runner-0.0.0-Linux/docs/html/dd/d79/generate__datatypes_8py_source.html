268 class StructType(Type):
269  def __init__(self, outname, xml, namespace):
270  Type.__init__(self, outname, xml, namespace)
271  self.members = []
272  lengthfields = [] # lengthfields of arrays are not included as members
273  for child in xml:
274  if child.get("LengthField"):
275  lengthfields.append(child.get("LengthField"))
276  for child in xml:
277  if not child.tag == "{http://opcfoundation.org/BinarySchema/}Field":
278  continue
279  if child.get("Name") in lengthfields:
280  continue
281  memberName = child.get("Name")
282  memberName = memberName[:1].lower() + memberName[1:]
283  memberTypeName = getTypeName(child.get("TypeName"))
284  memberType = types[memberTypeName]
285  isArray = True if child.get("LengthField") else False
286  self.members.append(StructMember(memberName, memberType, isArray))
287 
288  self.pointerfree = "true"
289  self.overlayable = "true"
290  self.kind = "UA_DATATYPEKIND_STRUCTURE"
291  before = None
292  for m in self.members:
293  if m.isArray or m.memberType.pointerfree != "true":
294  self.pointerfree = "false"
295  self.overlayable = "false"
296  else:
297  self.overlayable += "\n\t\t && " + m.memberType.overlayable
298  if before:
299  self.overlayable += "\n\t\t && offsetof(UA_%s, %s) == (offsetof(UA_%s, %s) + sizeof(UA_%s))" % \
300  (makeCIdentifier(self.name), makeCIdentifier(m.name), makeCIdentifier(self.name), makeCIdentifier(before.name), makeCIdentifier(before.memberType.name))
301  if "false" in self.overlayable:
302  self.overlayable = "false"
303  before = m
304 
305  def typedef_h(self):
306  if len(self.members) == 0:
307  return "typedef void * UA_%s;" % makeCIdentifier(self.name)
308  returnstr = "typedef struct {\n"
309  for member in self.members:
310  if member.isArray:
311  returnstr += " size_t %sSize;\n" % makeCIdentifier(member.name)

321 def parseTypeDefinitions(outname, xmlDescription, namespace, addToTypes=None):
322  def typeReady(element):
323  "Are all member types defined?"
324  for child in element:
325  if child.tag == "{http://opcfoundation.org/BinarySchema/}Field":
326  childname = getTypeName(child.get("TypeName"))
327  if childname not in types:
328  return False
329  return True
330 
331  def unknownTypes(element):
332  "Return all unknown types"
333  unknowns = []
334  for child in element:
335  if child.tag == "{http://opcfoundation.org/BinarySchema/}Field":
336  childname = getTypeName(child.get("TypeName"))
337  if childname not in types:
338  unknowns.append(childname)
339  return unknowns
340 
341  def skipType(name):
342  if name in excluded_types:
343  return True
344  if re.search("NodeId$", name) != None:
345  return True
346  return False
347 
348  snippets = {}
349  for typeXml in etree.parse(xmlDescription).getroot():
350  if not typeXml.get("Name"):
351  continue
352  name = typeXml.get("Name")
353  snippets[name] = typeXml
354 
355  detectLoop = len(snippets)+1
356  while(len(snippets) > 0):
357  if detectLoop == len(snippets):
358  name, typeXml = (snippets.items())[0]
359  raise RuntimeError("Infinite loop detected trying to processing types " + name + ": unknonwn subtype " + str(unknownTypes(typeXml)))
360  detectLoop = len(snippets)
361  for name, typeXml in list(snippets.items()):
362  if name in types or skipType(name):
363  del snippets[name]
364  continue
365  if not typeReady(typeXml):
366  continue
367  if name in builtin_types:
368  newType = BuiltinType(name)
369  elif typeXml.tag == "{http://opcfoundation.org/BinarySchema/}EnumeratedType":
370  newType = EnumerationType(outname, typeXml, namespace)
371  elif typeXml.tag == "{http://opcfoundation.org/BinarySchema/}OpaqueType":
372  newType = OpaqueType(outname, typeXml, namespace, get_base_type_for_opaque(name)['name'])
373  elif typeXml.tag == "{http://opcfoundation.org/BinarySchema/}StructuredType":
374  newType = StructType(outname, typeXml, namespace)
375  else:
376  raise Exception("Type not known")
377 
378  types[name] = newType
379  if addToTypes is not None:
380  addToTypes[name] = newType
381 
382  del snippets[name]

396 def parseTypeDescriptions(f, namespaceid):
397  definitions = {}
398 
399  csvreader = csv.reader(f, delimiter=',')
400  delay_init = []
401 
402  for row in csvreader:
403  if len(row) < 3:
404  continue
405  if row[2] == "Object":
406  # Check if node name ends with _Encoding_(DefaultXml|DefaultBinary) and store the node id in the corresponding DataType
407  m = re.match('(.*?)_Encoding_Default(Xml|Binary)$',row[0])
408  if (m):
409  baseType = m.group(1)
410  if baseType not in types:
411  continue
412 
413  delay_init.append({
414  "baseType": baseType,
415  "encoding": m.group(2),
416  "id": row[1]
417  })
418  continue
419  if row[2] != "DataType":
420  continue
421  if row[0] == "BaseDataType":
422  definitions["Variant"] = TypeDescription(row[0], row[1], namespaceid)
423  elif row[0] == "Structure":
424  definitions["ExtensionObject"] = TypeDescription(row[0], row[1], namespaceid)
425  elif row[0] not in types:
426  continue
427  else:
428  definitions[row[0]] = TypeDescription(row[0], row[1], namespaceid)
429  for i in delay_init:
430  if i["baseType"] not in definitions:
431  raise Exception("Type {} not found in definitions file.".format(i["baseType"]))
432  if i["encoding"] == "Xml":
433  definitions[i["baseType"]].xmlEncodingId = i["id"]
434  else:
435  definitions[i["baseType"]].binaryEncodingId = i["id"]
436  return definitions
