63 def extractNamespaces(xmlfile):
64  # Extract a list of namespaces used. The first namespace is always
65  # "http://opcfoundation.org/UA/". minidom gobbles up
66  # <NamespaceUris></NamespaceUris> elements, without a decent way to reliably
67  # access this dom2 <uri></uri> elements (only attribute xmlns= are accessible
68  # using minidom). We need them for dereferencing though... This function
69  # attempts to do just that.
70 
71  namespaces = ["http://opcfoundation.org/UA/"]
72  infile = codecs.open(xmlfile.name, encoding='utf-8')
73  foundURIs = False
74  nsline = ""
75  for line in infile:
76  if "<namespaceuris>" in line.lower():
77  foundURIs = True
78  elif "</namespaceuris>" in line.lower():
79  nsline = nsline + line
80  break
81  if foundURIs:
82  nsline = nsline + line
83 
84  if len(nsline) > 0:
85  ns = dom.parseString(nsline).getElementsByTagName("NamespaceUris")
86  for uri in ns[0].childNodes:
87  if uri.nodeType != uri.ELEMENT_NODE:
88  continue
89  if uri.firstChild.data in namespaces:
90  continue
91  namespaces.append(uri.firstChild.data)
92  infile.close()
93  return namespaces
